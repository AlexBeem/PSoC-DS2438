57 uint8_t DS2438_ReadRawRom(uint8_t* rom)
58 {
59  // Reset sequence
60  if (OneWire_TouchReset(DS2438_Pin_0) == 0)
61  {
62  // Write read rom command
63  OneWire_WriteByte(DS2438_Pin_0, DS2438_READ_ROM);
64  // Read 8 bytes of rom
65  uint8_t loop;
66  for (loop = 0; loop < 8; loop++)
67  {
68  rom[loop] = OneWire_ReadByte(DS2438_Pin_0);
69  }
70  if (crc_enabled == DS2438_DO_CRC_CHECK)
71  {
72  if (DS2438_CheckCrcValue(rom, 7, rom[7]) != DS2438_OK)
73  return DS2438_CRC_FAIL;
74  }
75  return DS2438_OK;
76  }
77  return DS2438_DEV_NOT_FOUND;
78 }

204 {
205  // Read page 0
206  uint8_t page_data[9];
207  uint8_t error = DS2438_ReadPage(0x00, page_data);
208  if (error == DS2438_OK)
209  {
210  // crc check
211  if (crc_enabled == DS2438_DO_CRC_CHECK)
212  {
213  if (DS2438_CheckCrcValue(page_data, 8, page_data[8]) != DS2438_OK)
214  {
215  return DS2438_CRC_FAIL;
216  }
217  }
218  // set bit based on input source in first byte of page 0

220  {
221  // set bit
222  page_data[0] = page_data[0] | (0x08);
223  }
224  else if (input_source == DS2438_INPUT_VOLTAGE_VAD)
225  {
226  // clear bit
227  page_data[0] = page_data[0] & (0xF7);
228  }
229  else
230  {
231  return DS2438_BAD_PARAM;
232  }
233  // write page 0
234  error = DS2438_WritePage(0x00, page_data);
235 
236  }
237  return error;

369 // Get current data in float format
370 uint8_t DS2438_GetCurrentData(float* current)
371 {
372  uint8_t page_data[9];
373  if (DS2438_ReadPage(0x00, page_data) == DS2438_OK)
374  {
375  if (crc_enabled == DS2438_DO_CRC_CHECK)
376  {
377  if (DS2438_CheckCrcValue(page_data, 8, page_data[8]) != DS2438_OK)
378  {
379  return DS2438_CRC_FAIL;
380  }
381  }
382  uint8_t curr_lsb = page_data[5];
383  uint8_t curr_msb = page_data[6];
384  int16_t curr_data = 0;
385  // Get if positive or negative
386  if ((curr_msb & 0x03) > 1)
387  {
388  // current is negative
389  uint16_t data = (curr_msb<<8) | (curr_lsb);
390  // perform 2's complement
391  int16_t temp = (~data) & 0x3FF;
392  curr_data = (short)(temp * -1);
393  }
394  else
395  {
396  curr_data = (curr_msb << 8) | (curr_lsb & 0xFF);
397  }
398 
399  *current = (curr_data) / (4096.*DS2438_SENSE_RESISTOR);
400  return DS2438_OK;
401  }
402  else
403  {
404  return DS2438_ERROR;
405  }
406  return DS2438_DEV_NOT_FOUND;
407 }

488 // Write current threshold value
489 uint8_t DS2438_WriteThreshold(uint8_t threshold)
490 {
491  if (threshold > 3)
492  return DS2438_BAD_PARAM;
493  // Threshold is located at byte 7 of page 0
494  uint8_t page_data[9];
495  uint8_t error = DS2438_ReadPage(0x00, page_data);
496  uint8_t IAD_active = 0;
497  if ( error == DS2438_OK)
498  {
499 
500  // Stop IAD
501  if (page_data[0] | 0x01)
502  {
503  DS2438_DisableIAD();
504  IAD_active = 1;
505  }
506 
507  // Update threshold bits
508  page_data[7] = threshold << 6;
509  // Write new page data
510  DS2438_WritePage(0x00, page_data);
511  if (IAD_active == 1)
512  {
513  DS2438_EnableIAD();
514  }
515  }
516  return error;
517 }

519 uint8_t DS2438_WriteOffset(int16_t offset)
520 {
521  // Offset is located at bytes 5-6 of page 1
522  uint8_t page_data[9];
523  uint8_t error = DS2438_ReadPage(0x01, page_data);
524  uint8_t IAD_active = 0;
525  if ( error == DS2438_OK)
526  {
527 
528  // Stop IAD
529  if (page_data[0] | 0x01)
530  {
531  DS2438_DisableIAD();
532  IAD_active = 1;
533  }
534  // Keep 5 LSBs and shift them to the left by 3
535  uint8_t offset_lsb = ( (offset << 3) & 0xF8);
536 
537  // Keep 5 MSBs and shift them to the right by 8
538  uint8_t offset_msb = ( (offset & 0x3E0) >> 8);
539  // Update threshold bits
540  page_data[5] = offset_lsb;
541  page_data[6] = offset_msb;
542  // Write new page data
543  DS2438_WritePage(0x00, page_data);
544  if (IAD_active == 1)
545  {
546  DS2438_EnableIAD();
547  }
548  }
549  return error;
550 }

673 // Read one page of data
674 uint8_t DS2438_ReadPage(uint8_t page_number, uint8_t* page_data)
675 {
676  if (page_number > 0x07)
677  return DS2438_BAD_PARAM;
678  else
679  {
680  // Reset sequence
681  if (OneWire_TouchReset(DS2438_Pin_0) == 0)
682  {
683  // Skip ROM command
684  OneWire_WriteByte(DS2438_Pin_0, DS2438_SKIP_ROM);
685  // Recall memory command
686  OneWire_WriteByte(DS2438_Pin_0, DS2438_RECALL_MEMORY);
687  // Page number
688  OneWire_WriteByte(DS2438_Pin_0, page_number);
689  if (OneWire_TouchReset(DS2438_Pin_0) == 0)
690  {
691  // Skip ROM command
692  OneWire_WriteByte(DS2438_Pin_0, DS2438_SKIP_ROM);
693  // Read scratchpad command
694  OneWire_WriteByte(DS2438_Pin_0, DS2438_READ_SCRATCHPAD);
695  // Page 0
696  OneWire_WriteByte(DS2438_Pin_0, page_number);
697  // Read nine bytes
698  for (uint8_t i = 0; i < 9; i++)
699  {
700  page_data[i] = OneWire_ReadByte(DS2438_Pin_0);
701  }
702  return DS2438_OK;
703 
704  }
705  }
706  }
707  return DS2438_DEV_NOT_FOUND;
708 }

760 uint8_t DS2438_ComputeCrc(const uint8_t *data, uint8_t len)
761 {
762  uint8_t crc = 0;
763 
764  while (len--) {
765 
766  uint8_t inbyte = *data++;
767  for (uint8_t i = 8; i; i--) {
768  uint8_t mix = (crc ^ inbyte) & 0x01;
769  crc >>= 1;

771  inbyte >>= 1;
772  }
