29 // Generate a 1-Wire reset, return 1 if no presence detect was found,
30 // return 0 otherwise.
31 int OneWire_TouchReset(unsigned int pin)
32 {
33  int result;
34 
35  CyDelayUs(DELAY_G);
36  CyPins_ClearPin(pin); // Drives DQ low
37  CyDelayUs(DELAY_H);
38  CyPins_SetPin(pin); // Releases the bus
39  CyDelayUs(DELAY_I);
40  result = CyPins_ReadPin(pin) > 0 ? 1 : 0; // Sample for presence pulse from slave
41  CyDelayUs(DELAY_J); // Complete the reset sequence recovery
42  return result; // Return sample presence pulse result
43 }
